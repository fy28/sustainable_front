<script>
import jsPDF from "jspdf";
import autoTable from "jspdf-autotable";

export default {
  name: "ExpeditionList",
  data() {
    return {
      expedition: [],
      globalList: [],
      showModal: false,
      newDocName: "",
      selectedProduct: "",
      clientName: "",
      totalQuantity: "",
      unit: "",
    };
  },
  computed: {
    productList() {
      return [...new Set(this.expedition.map((e) => e.product))];
    },
  },
  created() {
    const saved = localStorage.getItem("expedition");
    this.expedition = saved ? JSON.parse(saved) : [];

    // ‚úÖ R√©cup√©ration des infos globales (client, quantit√©, unit√©)
if (this.expedition.length > 0) {
  const first = this.expedition[0];
  this.clientName = first.clientName || this.$route.params.clientName || "Client inconnu";

  // üîπ Regroupement par produit + unit√© + pays
  const grouped = this.expedition.reduce((acc, exp) => {
    const product = exp.product || "Produit inconnu";
    const unit = exp.unit || "kg";
    const country = exp.country || "pays inconnu";
    const qty = parseFloat(exp.quantity) || 0;

    const key = `${product}|${unit}|${country}`;
    acc[key] = (acc[key] || 0) + qty;
    return acc;
  }, {});

  // üîπ Format lisible ‚Üí "450 kg de vanille (france) + 1 tonne de vanille (am√©rique)"
  const formatted = Object.entries(grouped)
    .map(([key, qty]) => {
      const [product, unit, country] = key.split("|");
      return `${qty} ${unit}${qty > 1 ? "s" : ""} de ${product} (${country})`;
    })
    .join(" + ");

  this.totalQuantity = formatted;
  this.unit = first.unit || "";
}


    // ‚úÖ Construction de la liste globale
    this.globalList = this.expedition.flatMap((exp) =>
      exp.documents.map((doc) => ({
        product: exp.product,
        country: exp.country,
        doc: doc,
        deliveryDate: exp.deliveryDate,
        file: null,
      }))
    );
  },
  methods: {
    formatDate(date) {
      if (!date) return "‚Äî";
      const d = new Date(date);
      return d.toLocaleDateString("fr-FR", {
        year: "numeric",
        month: "short",
        day: "numeric",
      });
    },
    confirmExpedition() {
      alert("‚úÖ Exp√©dition enregistr√©e !");
    },
    sendEmail() {
      alert("üìß Email envoy√© au client !");
    },
    triggerFileInput(index) {
      this.$refs.fileInputs[index].click();
    },
    handleFileUpload(event, index) {
      const file = event.target.files[0];
      if (!file) return;
      const blobUrl = URL.createObjectURL(file);
      this.globalList[index].file = blobUrl;
      this.globalList[index].fileName = file.name;
    },
    deleteDocument(index) {
      if (confirm("Supprimer ce document ?")) this.globalList.splice(index, 1);
    },
    exportPDF() {
      const pdf = new jsPDF({ orientation: "portrait", unit: "mm", format: "a4" });
      pdf.setFontSize(16);
      pdf.text("Liste des documents d√©finitifs", 14, 20);
      pdf.setFontSize(11);
      pdf.text(`Client : ${this.clientName}`, 14, 28);
      pdf.text(`Quantit√© : ${this.totalQuantity} ${this.unit}`, 14, 34);

      const tableData = this.globalList.map((item, index) => [
        index + 1,
        item.doc,
        item.product,
        item.country,
        this.formatDate(item.deliveryDate),
        item.file ? "Fourni" : "Non fourni",
      ]);

      autoTable(pdf, {
        startY: 40,
        head: [["#", "Document", "Produit", "Pays", "Livraison", "Statut"]],
        body: tableData,
        theme: "grid",
        styles: { halign: "center", valign: "middle", fontSize: 9 },
        headStyles: { fillColor: [60, 141, 188], textColor: 255, fontStyle: "bold" },
        alternateRowStyles: { fillColor: [245, 245, 245] },
      });

      pdf.save("expedition-documents.pdf");
    },
    confirmAddDocument() {
      if (!this.newDocName || !this.selectedProduct) {
        alert("Veuillez remplir tous les champs !");
        return;
      }

      if (
        confirm(
          "Souhaitez-vous ajouter ce document pour TOUTES les exp√©ditions √† venir ?\nCliquez sur Annuler pour l‚Äôajouter seulement √† celle-ci."
        )
      ) {
        alert("‚úÖ Ce document sera ajout√© pour les futures exp√©ditions (fonctionnalit√© √† venir).");
      } else {
        const expToUpdate = this.expedition.find((e) => e.product === this.selectedProduct);
        if (expToUpdate) {
          this.globalList.push({
            product: expToUpdate.product,
            country: expToUpdate.country,
            doc: this.newDocName,
            deliveryDate: expToUpdate.deliveryDate,
            file: null,
          });
        }
      }

      this.newDocName = "";
      this.selectedProduct = "";
      this.showModal = false;
    },
  },
};
</script>

<script>
const DOCUMENTS_RULES = {
  vanille: {
    france: ["Phytosanitaire", "Certification Bio", "Facture commerciale"],
    amerique: ["Phytosanitaire", "Num√©ro lock", "Licence d‚Äôexportation"],
  },
  girofle: {
    france: ["Facture commerciale", "D√©claration en douane", "Licence d‚Äôexportation"],
    amerique: ["Facture commerciale", "Analyse qualit√©", "Licence d‚Äôexportation"],
  },
};

export default {
  name: "NewExpedition",
  data() {
    return {
      products: [], // üîÑ sera rempli dynamiquement depuis localStorage
      clientCountries: ["france", "amerique"],

      selectedProduct: "",
      selectedCountry: "", // ‚úÖ un seul pays
      deliveryDate: "",
      quantity: "",
      unit: "kg",

      expedition: [],
      showAddForm: false,
    };
  },
  created() {
    // üîπ Charger les produits depuis localStorage
    const savedProducts = JSON.parse(localStorage.getItem("produits") || "[]");

    if (savedProducts.length === 0) {
      this.products = Object.keys(DOCUMENTS_RULES).map((k) => ({
        nom: k,
        codeHS: "",
        prix: 0,
        unite: "kg",
        documents: [],
      }));
    } else {
      this.products = savedProducts;
    }
  },
  computed: {
    selectedProductDetails() {
      return this.products.find((p) => p.nom === this.selectedProduct);
    },
  },
  methods: {
    addExport() {
      if (!this.selectedProduct || !this.selectedCountry) {
        alert("Veuillez choisir un produit et un pays de destination !");
        return;
      }

      // üîπ R√®gles automatiques selon DOCUMENTS_RULES ou produit personnalis√©
      const selectedProd = this.products.find((p) => p.nom === this.selectedProduct);
      const docs =
        selectedProd?.documents?.length > 0
          ? selectedProd.documents
          : DOCUMENTS_RULES[this.selectedProduct?.toLowerCase()]?.[this.selectedCountry] ||
            ["Facture commerciale", "Licence d‚Äôexportation"];

      this.expedition.push({
        product: this.selectedProduct,
        country: this.selectedCountry,
        documents: docs,
        deliveryDate: this.deliveryDate,
        quantity: this.quantity,
        unit: this.unit,
      });

      // reset
      this.selectedProduct = "";
      this.selectedCountry = "";
      this.deliveryDate = "";
      this.quantity = "";
      this.unit = "kg";
      this.showAddForm = false;
    },
    removeProduct(index) {
      this.expedition.splice(index, 1);
    },
    validateExpedition() {
      const clientName = this.$route.params.clientName;
      const clientId = this.$route.params.clientId;

      const enrichedExpedition = this.expedition.map((exp) => ({
        ...exp,
        clientName,
        clientId,
      }));

      localStorage.setItem("expedition", JSON.stringify(enrichedExpedition));

      this.$router.push({ name: "ExpeditionList" });
    },
  },
};
</script>
